[docs]def is_bag(file_name):
    """ Determine if a file is valid BAG (False if it doesn't exist). """

    # we first check if the file is a valid hdf5 (it also checks if the file exists)
    if not h5py.is_hdf5(file_name):
        return False

    fid = h5py.File(file_name, 'r')

    try:
        fid["BAG_root"]

    except KeyError:
        return False

    return True

        """
        Create a new file object.

    @with_phil
    def __repr__(self):
        if not self.id:
            log.info("closed file")
            r = u('<Closed BAG file>')
        else:
            # Filename has to be forced to Unicode if it comes back bytes
            # Mode is always a "native" string
            filename = self.filename
            if isinstance(filename, bytes):  # Can't decode fname
                filename = filename.decode('utf8', 'replace')
            r = u('<BAG file "%s" (mode %s)>') % (os.path.basename(filename), self.mode)
            r += "\n .id: %s" % self.id
            r += "\n .name: %s" % self.name
            r += "\n .driver: %s" % self.driver
            r += "\n .user block size: %s" % self.userblock_size

        if PY3:
            return r
        return r.encode('utf8')
