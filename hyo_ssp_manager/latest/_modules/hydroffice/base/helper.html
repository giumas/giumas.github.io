[docs]class HyOError(Exception):
    """ Base class for exceptions in HydrOffice """

    def __init__(self, message, *args):
        if isinstance(message, Exception):
            msg = message.args[0] if len(message.args) > 0 else ''
        else:
            msg = message

        self.message = msg
        # allow users initialize misc. arguments as any other builtin Error
        Exception.__init__(self, message, *args)

        """ copy anything from src to dst """
        import errno
        try:
            shutil.copytree(src, dst)
        except OSError as exc:  # python >2.5
            if exc.errno == errno.ENOTDIR:
                shutil.copy(src, dst)
            else:
                raise IOError("unable to copy from %s to %s" % (src, dst))

[docs]    def default_projects_folder():
        """ Function that return the default project folder for HydrOffice

        On Windows, if still not present, it create a folder "HydrOffice X.Y.Z" on the user's doc folder
        On Linux and Mac, it creates the name folder into the user's Home directory
        """

        # first we retrieve the base folder that is related to the current user and OS
        # then we add the HydrOffice specific part
        if Helper.is_windows():
            base_folder = os.getenv('USERPROFILE')
            base_folder = os.path.join(base_folder, "Documents")
        else:  # linux and MacOs
            base_folder = os.getenv('HOME')

        projects_folder = os.path.join(base_folder, "HydrOffice %s" % hyo_base_version())

        if not os.path.exists(projects_folder):
            try:
                os.mkdir(projects_folder)
            except OSError as e:
                raise HyOError("Error in creating the default projects folder: %s\n"
                               "%s" % (projects_folder, e))

        return projects_folder

[docs]    def get_file_properties(file_name):
        """
        Read all properties of the given file return them as a dictionary. [only for Windows]
        """
        if not Helper.is_windows():
            raise EnvironmentError("Calling a Windows-only method")

        import win32api

        prop_names = ('Comments', 'InternalName', 'ProductName', 'CompanyName', 'LegalCopyright',
                      'ProductVersion', 'FileDescription', 'LegalTrademarks', 'PrivateBuild',
                      'FileVersion', 'OriginalFilename', 'SpecialBuild')

        props = {'FixedFileInfo': None, 'StringFileInfo': None, 'FileVersion': None}

        # try:
        # backslash as parm returns dictionary of numeric info corresponding to VS_FIXEDFILEINFO struc
        fixed_info = win32api.GetFileVersionInfo(file_name, '\\')
        props['FixedFileInfo'] = fixed_info
        props['FileVersion'] = "%d.%d.%d.%d" % (fixed_info['FileVersionMS'] / 65536,
                                                fixed_info['FileVersionMS'] % 65536,
                                                fixed_info['FileVersionLS'] / 65536,
                                                fixed_info['FileVersionLS'] % 65536)

        # \VarFileInfo\Translation returns list of available (language, code_page)
        # pairs that can be used to retreive string info. We are using only the first pair.
        lang, code_page = win32api.GetFileVersionInfo(file_name, '\\VarFileInfo\\Translation')[0]

        # any other must be of the form \StringfileInfo\%04X%04X\parm_name, middle
        # two are language/code_page pair returned from above

        str_info = {}
        for prop_name in prop_names:
            str_info_path = "\\StringFileInfo\\%04X%04X\\%s" % (lang, code_page, prop_name)
            # print str_info
            str_info[prop_name] = win32api.GetFileVersionInfo(file_name, str_info_path)

        props['StringFileInfo'] = str_info

        return props

        """
        Check the python version required by the passed pyd library [only for Windows]
        e.g. "C:\\Program Files\\CARIS\\BDB\\4.1\\python\\caris\\coverage\\_init_caris_python.pyd"
        e.g. "C:\\py34x64\\Lib\\site-packages\\win32\\win32api.pyd"

        if not Helper.is_windows():
            raise EnvironmentError("Calling a Windows-only method")

        import win32api

        print("- pyd path: %s" % str(pyd_path))
        try:
            info = win32api.GetFileVersionInfo(pyd_path, '\\')
        except win32api.error as e:
            # print(dir(e))
            print(str(e.strerror) + " > " + pyd_path)
            return ""

        # print(info)
        python_version = "%s.%s.%s.%s" % (
            str(win32api.HIWORD(info['FileVersionMS'])),
            str(win32api.LOWORD(info['FileVersionMS'])),
            str(win32api.HIWORD(info['FileVersionLS'])),
            str(win32api.LOWORD(info['FileVersionLS'])))
        return python_version

[docs]    def get_separator(aline, headers):
        """ Try to determine what separator is used between D, M, and S, for example, a space or a dash? """
        sep_char = '-'
        try:
            float(aline.split('|')[headers['latitude']].split(sep_char)[0])
        except NameError:
            raise HyOError('NameError parsing %s with headers %s and separator \'%s\''
                           % (aline, headers, sep_char))
        except IndexError:
            raise HyOError('IndexError parsing %s with headers %s and separator \'%s\''
                           % (aline, headers, sep_char))
        except ValueError:
            sep_char = ' '
            try:
                float(aline.split('|')[headers['latitude']].split(sep_char)[0])
            except NameError:
                raise HyOError('NameError parsing %s with headers %s and separator \'%s\''
                               % (aline, headers, sep_char))
            except IndexError:
                raise HyOError('IndexError parsing %s with headers %s and separator \'%s\''
                               % (aline, headers, sep_char))
            except ValueError:
                raise HyOError('ValueError parsing %s with headers %s and separator \'%s\''
                               % (aline, headers, sep_char))

        return sep_char

[docs]    def noaa_sounding_rounding(depth_in_meters):
        """
        NOAA sounding rounding convention for chart display: all depths less than 4.75 feet are
        rounded down to 4 feet, depths greater than 4.75 feet are rounding up to 5 feet.
        """
        depth_in_feet = depth_in_meters / 0.3048
        remainder = depth_in_feet - np.floor(depth_in_feet)
        if remainder < 0.75:
            rounded_depth = np.floor(depth_in_feet)
        else:
            rounded_depth = round(depth_in_feet)
        return rounded_depth

[docs]    def list_hydro_packages():
        """ list all the available hydro-packages """
        import pkgutil
        import hydroffice

        hyo_pkgs = []

        for pkg_loader, pkg_name, is_pkg in pkgutil.walk_packages(hydroffice.__path__, hydroffice.__name__ + '.'):
            if not is_pkg:
                continue
            if "hydroffice.base" in pkg_name:
                continue

            # load the module
            pkg = pkg_loader.find_module(pkg_name).load_module(pkg_name)
            try:
                pkg_info = pkg.hyo()
            except AttributeError:  # the HydrOffice ID function does not exist
                continue

            hyo_pkgs.append((pkg_name, pkg_loader, pkg_info))

        return hyo_pkgs

[docs]    def explore_folder(cls, path):
        """
        Open the passed path using OS-native commands
        """
        if not path:
            return

        import subprocess

        if cls.is_darwin():
            subprocess.call(['open', '--', path])

        elif cls.is_linux():
            subprocess.call(['xdg-open', path])

        elif cls.is_windows():
            subprocess.call(['explorer', path])

        else:
            raise OSError("Unknown/unsupported OS")
