[docs]def check_mandatory_attributes(s57list, objects, attributes, report=None):
    # report section
    if report:
        report << "Mandatory attributes [CHECK]"

    info = list()
    for m in range(0, len(s57list)):
        toggle = 0
        if s57list[m].acronym in objects:
            flag = ''
            for n in range(0, len(attributes)):
                if not hasattr(s57list[m], attributes[n]):
                    toggle = 1
                    flag = flag + attributes[n] + ', '
            if toggle == 1 and flag[-2] == ',':
                flag = flag[0:-2]
            if toggle == 1 and len(flag) > 0:
                if hasattr(s57list[m], 'latitude'):
                    print(s57list[m].acronym + ' at ' + str(round(s57list[m].latitude, 5)) + ', ' +
                          str(round(s57list[m].longitude, 5)) + ' is missing ' + flag)
                    info.append([s57list[m].latitude, s57list[m].longitude, flag])
                else:
                    if report:
                        report << ' %s line/area feature is missing %s' % (s57list[m].acronym, flag)
                    else:
                        print(s57list[m].acronym + ' line/area feature is missing ' + flag)

    if report and (len(info) == 0):
        report << "OK"

    return info

    # report section
    if report:
        report << "Feature redundancy [CHECK]"

    features = list()
    for m in range(0, len(s57list)):
        if hasattr(s57list[m], 'latitude') and hasattr(s57list[m], 'longitude'):
            features.append([s57list[m].latitude, s57list[m].longitude, s57list[m].acronym])
    iredundancy = list()
    for m in range(0, len(features)):
        i = features.count(features[m])
        if i > 1:
            iredundancy.append([features[m][0], features[m][1], 'redundant ' + features[m][2]])
    redundancy = list()
    for m in range(0, len(iredundancy)):
        if iredundancy[m] not in redundancy:
            if report:
                report << 'found %s at %s, %s' % \
                          (iredundancy[m][2], str(round(iredundancy[m][0], 5)), str(round(iredundancy[m][1], 5)))
            else:
                print(' - found ' + iredundancy[m][2] + ' at ' + str(round(iredundancy[m][0], 5)) +
                      ', ' + str(round(iredundancy[m][1], 5)))
            redundancy.append(iredundancy[m])

    if report and (len(redundancy) == 0):
        report << "OK"

    return redundancy

[docs]def check_s57_linelist(linelist):
    merged = list()
    merged.append(linelist[0])
    base = len(linelist[0].split('|'))
    toggle = 0
    for m in range(1, len(linelist)):
        if len(linelist[m].split('|')) < base and toggle == 0:
            merged.append(linelist[m])
            toggle = 1
        elif len(linelist[m].split('|')) < base and toggle == 1:
            merged[-1] = merged[-1] + linelist[m]
        else:
            merged.append(linelist[m])
            toggle = 0
    return merged

    def __init__(self, aline, headers):
        """ Identify field in the line using the headers dictionary """

        splits = aline.split('|')

        self.acronym = splits[headers['acronym']]
        self.geometry = splits[headers['geometry']]

        if len(splits[headers['depth']]) > 0:
            depth = splits[headers['depth']]
            if self.geometry == 'Sounding':
                self.depth = float(depth)
            else:
                self.depth = depth

        if len(splits[headers['valsou']]) > 0:
            valsou = splits[headers['valsou']]

                    self.valsou = valsou
