[docs]    def check_tables_values_in_cols(self, table_name, print_out=False):
        """ Returns a dictionary with columns as keys and the number of not-null entries

        Args:
            table_name (str):      the name of the table to check
            print_out (bool):      option to print out the result of the query
        Returns:
            dict:                  dictionary with columns as keys and the number of not-null entries
        """
        table_name = BaseDbObject.clean_name(table_name)

        col_dict = dict()
        for ret in self.conn.execute('PRAGMA TABLE_INFO({})'.format(table_name)):
            col_dict[ret[1]] = 0

        for col in col_dict:
            for ret2 in self.conn.execute('SELECT COUNT({0}) FROM {1} WHERE {0} IS NOT NULL'.format(col, table_name)):
                col_dict[col] = ret2[0]

        if print_out:
            print("\nNumber of entries per column:")
            for i in col_dict.items():
                print('{}: {}'.format(i[0], i[1]))

        return col_dict



[docs]    def reconnect_or_create(self):
        """ Reconnection to an existing database or create a new db """
        if self.conn:
            log.info("Already connected")

        if not os.path.exists(self.db_path):
            log.info("New db")

        try:
            self.conn = sqlite3.connect(self.db_path,
                                        detect_types=sqlite3.PARSE_DECLTYPES | sqlite3.PARSE_COLNAMES)
            log.info("Connected")

        except sqlite3.Error as e:
            raise HyOError("Unable to connect: %s" % e)

        try:
            with self.conn:
                self.conn.execute('PRAGMA foreign_keys=ON')
                self.conn.commit()

        except sqlite3.Error as e:
            raise HyOError("Unable to activate foreign keys: %s" % e)

        try:
            # Set the row factory
            self.conn.row_factory = sqlite3.Row
            # Register the adapter
            sqlite3.register_adapter(BaseDbObject.Point, BaseDbObject.adapt_point)
            # Register the converter
            sqlite3.register_converter(b"point", BaseDbObject.convert_point)

        except sqlite3.Error as e:
            raise HyOError("Unable to register 'point': %s - %s" % (type(e), e))

        built = self.build_tables()
        if not isinstance(built, bool):
            raise HyOError("invalid return from 'build_tables' method, must be boolean")
        if not built:
            raise HyOError("Unable to build tables: the DB is encrypted or is not a database")

[docs]    def disconnect(self):
        """ Disconnect from the current database """
        if not self.conn:
            log.info("Already disconnected")
            return True

        try:
            self.conn.close()
            log.info("Disconnected")
            return True

        except sqlite3.Error as e:
            log.error("Unable to disconnect: %s" % e)
            return False
