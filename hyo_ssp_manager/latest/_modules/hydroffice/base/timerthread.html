[docs]class TimerThread(threading.Thread):
    """ Thread sub-class to implement a single-thread timer that can be stopped """

    def __init__(self, target, timing=1):
        threading.Thread.__init__(self)
        self.event = threading.Event()
        self.target = target
        self.timing = timing


[docs]    def run(self):
        next_call = time.time()
        while not self.event.is_set():
            next_call += self.timing
            self.target()
            self.event.wait(next_call - time.time())
